# Template: k6 Load Test Script

Source: generated by `/load-test` skill
Pattern: staged load test with thresholds and checks

```javascript
// load-test.js
import http from 'k6/http';
import { check, group, sleep } from 'k6';
import { Rate, Trend, Counter } from 'k6/metrics';

// Custom metrics
const errorRate = new Rate('errors');
const apiDuration = new Trend('api_duration');
const requestCount = new Counter('api_requests');

export const options = {
  stages: [
    { duration: '30s', target: 20 },    // ramp up to 20 users
    { duration: '1m', target: 20 },     // stay at 20 users
    { duration: '30s', target: 50 },    // ramp to 50 users
    { duration: '1m', target: 50 },     // stay at 50 users
    { duration: '30s', target: 0 },     // ramp down
  ],
  thresholds: {
    http_req_duration: ['p(95)<500', 'p(99)<1500'],
    http_req_failed: ['rate<0.01'],
    errors: ['rate<0.05'],
  },
};

const BASE_URL = __ENV.BASE_URL || 'http://localhost:{port}';

export function setup() {
  // Login to get auth token (if needed)
  const loginRes = http.post(`${BASE_URL}/api/v1/auth/login`, JSON.stringify({
    email: 'loadtest@test.com',
    password: 'Test1234!',
  }), { headers: { 'Content-Type': 'application/json' } });

  return { token: loginRes.json('access_token') };
}

export default function (data) {
  const headers = {
    'Content-Type': 'application/json',
    Authorization: `Bearer ${data.token}`,
  };

  group('Health Check', () => {
    const res = http.get(`${BASE_URL}/health`);
    check(res, { 'health 200': (r) => r.status === 200 });
  });

  group('API Endpoints', () => {
    const res = http.get(`${BASE_URL}/api/v1/{endpoint}`, { headers });
    check(res, {
      'status 200': (r) => r.status === 200,
      'response < 500ms': (r) => r.timings.duration < 500,
    });
    apiDuration.add(res.timings.duration);
    requestCount.add(1);
    errorRate.add(res.status !== 200);
  });

  sleep(1);
}

export function teardown(data) {
  // Cleanup test data if needed
}
```

## Usage
Use when setting up load testing via `/load-test`.

## Customization Points
- `{port}`: API port (e.g., 8000)
- `{endpoint}`: Target API endpoint
- `stages`: Adjust load profile (users, duration)
- `thresholds`: Set acceptable performance limits
- `setup()`: Configure authentication
