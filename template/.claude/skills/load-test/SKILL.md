## Tool 1: k6 (Recommended)

### Generate Script (if none exists)
```javascript
// load-test.js — auto-generated by /load-test
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Rate, Trend } from 'k6/metrics';

const errorRate = new Rate('errors');
const responseTime = new Trend('response_time');

export const options = {
  stages: [
    { duration: '10s', target: 10 },   // ramp up
    { duration: '30s', target: 50 },   // sustained load
    { duration: '10s', target: 100 },  // peak load
    { duration: '10s', target: 0 },    // ramp down
  ],
  thresholds: {
    http_req_duration: ['p(95)<500', 'p(99)<1000'],
    errors: ['rate<0.01'],
    http_req_failed: ['rate<0.01'],
  },
};

const BASE_URL = __ENV.BASE_URL || 'http://localhost:8000';

export default function () {
  // Health check
  const healthRes = http.get(`${BASE_URL}/health`);
  check(healthRes, { 'health OK': (r) => r.status === 200 });

  // API endpoints (customize per project)
  const apiRes = http.get(`${BASE_URL}/api/v1/{endpoint}`);
  check(apiRes, {
    'status 200': (r) => r.status === 200,
    'response < 500ms': (r) => r.timings.duration < 500,
  });
  responseTime.add(apiRes.timings.duration);
  errorRate.add(apiRes.status !== 200);

  sleep(1);
}
```

### Execute k6
```bash
mkdir -p .claude/reports/load

# Run with JSON output for parsing
k6 run load-test.js \
  --out json=.claude/reports/load/k6-raw.json \
  --summary-export=.claude/reports/load/k6-summary.json \
  -e BASE_URL="${TARGET:-http://localhost:8000}" \
  ${VUS:+--vus $VUS} \
  ${DURATION:+--duration $DURATION} \
  2>&1 | tee .claude/reports/load/k6-output.txt

# Parse summary
cat .claude/reports/load/k6-summary.json | python3 -m json.tool
```
